and only if 2**p-1 divides S(p-1) where S(n+1)=S(n)**2-2, and S(1)=4.

INT p = 11213; 

printf (($l"M_", g(0), b(" is", " is not"), " prime"$, p, 
         (
           LONG LONG INT m p =  LONG LONG 2 ** p - 1, LONG LONG INT s := 4;
           FROM 3 TO p
           DO s := (s * s - 2) MOD m p
           OD;
           s = 0
         ) 
       ))
