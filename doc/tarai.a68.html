# 
Takeuchi's Tarai (or Tak) function. Moore proved its termination. 
See http://mathworld.wolfram.com/TAKFunction.html
#

PROC tarai = (INT i, j, k) INT:
     IF i <= j 
     THEN j
     ELSE tarai (tarai (i - 1, j, k),
                 tarai (j - 1, k, i),
                 tarai (k - 1, i, j)) 
     FI;

tarai (1, 2, 3)
