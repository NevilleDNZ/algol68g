# After MC Algol 68 Test Set decl03 #

BEGIN # P.G. Hibbard, Proc. Int. Conf. A68 III, Winnipeg, June, 1974:
        applied occurrence of mode-indication in actual-bounds
        of its actual-declarer #

      INT n := 4, CHAR a := "a", b := "b", c := "c";

      PROC swap = (REF CHAR c1, c2) VOID: (CHAR d = c1; c1 := c2; c2 := d);

      MODE HANOI =
         (IF n > 0
          THEN n -:= 1; 
	       swap (b, c); 
	       HANOI h1; 
	       swap (b, c);
               print ((newline, "move ", whole(LWB h1 + 1, -1), " from ", a, " to ", c, "."));
               swap (a, b);
	       VOID (LOC HANOI);
	       swap (a, b);
	       n +:= 1
          ELSE 0
          FI .. 1) INT;

      VOID (LOC HANOI)
END
